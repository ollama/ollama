118 def decode_token(self, token: str) -> dict[str, Any]: 

119 """Decode and validate JWT token.

121 Args: 

122 token: Encoded JWT token 

123 

124 Returns: 

125 Token payload 

126 

127 Raises: 

128 AuthenticationError: If token is invalid or expired 

129 """ 

130 try: 

131 payload = jwt.decode(token, self.secret_key, algorithms=[self.algorithm]) 

132 return cast(dict[str, Any], payload)
