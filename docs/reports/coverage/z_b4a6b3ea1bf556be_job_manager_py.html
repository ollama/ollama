86 async def cancel_job(self, job_id: UUID) -> bool: 

87 """Attempt to cancel a running or pending training job. 

88 

89 Args: 

90 job_id: Unique identifier for the job. 

91 

92 Returns: 

93 True if cancellation was initiated, False otherwise. 

94 """ 

95 log.info("cancelling_training_job", job_id=str(job_id)) 

96 

97 db_job = await self.repo.get_by_id(job_id) 

98 if not db_job: 

99 log.warning("cancel_requested_not_found", job_id=str(job_id)) 

100 return False 

101 

102 current_status = TrainingStatus(db_job.status) 

103 

104 # If already terminal, nothing to do 

105 if current_status in ( 

106 TrainingStatus.COMPLETED, 

107 TrainingStatus.FAILED, 

108 TrainingStatus.CANCELLED, 

109 ): 

110 log.info("cancel_ignored_already_terminal", job_id=str(job_id), status=current_status) 

111 return False 

112 

113 # If it's currently training, try to signal the engine 

114 if current_status == TrainingStatus.TRAINING and self.engine: 

115 await self.engine.stop(str(job_id)) 

116 

117 # Update database status 

118 await self.repo.update_status(job_id, TrainingStatus.CANCELLED) 

119 # Note: caller is responsible for commit if they handle the transaction 

120 # But here repository.update_status might need a session.commit() 

121 

122 return True
