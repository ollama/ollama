54 async def list_models(self) -> list[Model]: 

55 """List all available models on Ollama server. 

56 

57 Returns: 

58 List of available models with metadata. 

59 

60 Raises: 

61 httpx.HTTPError: If API request fails. 

62 """ 

63 log.info("ollama_list_models", endpoint=self.base_url) 

64 

65 response = await self.client.get("/api/tags") 

66 response.raise_for_status() 

67 

68 data = response.json() 

69 models: list[Model] = [ 

70 Model( 

71 name=m.get("name", "unknown"), 

72 size=str(m.get("size", "")), 

73 model_type=ModelType.TEXT_GENERATION, 

74 description=m.get("details", {}).get("description", ""), 

75 parameters=int(m.get("parameters", 0) or 0), 

76 context_length=int(m.get("context_length", 0) or 0), 

77 quantization=m.get("quantization", ""), 

78 loaded=bool(m.get("loaded", False)), 

79 ) 

80 for m in data.get("models", []) 

81 ] 

82 

83 log.info("ollama_models_listed", count=len(models)) 

84 return models

113 async def generate_stream( 

114 self, request: GenerateRequest 

115 ) -> AsyncGenerator[GenerateResponse, None]: 

116 """Generate text completion with streaming. 

117 

118 Args: 

119 request: Generation request with prompt and parameters. 

120 

121 Yields: 

122 Generated response chunks. 

123 """ 

124 log.info("ollama_generate_stream", model=request.model, prompt_len=len(request.prompt)) 

125 

126 payload: dict[str, Any] = { 

127 "model": request.model, 

128 "prompt": request.prompt, 

129 "temperature": request.temperature, 

130 "top_p": request.top_p, 

131 "top_k": request.top_k, 

132 "num_predict": request.num_predict, 

133 "stop": request.stop, 

134 "stream": True, 

135 } 

136 

137 async with self.client.stream("POST", "/api/generate", json=payload) as response: 

138 response.raise_for_status() 

139 async for line in response.aiter_lines(): 

140 if not line: 

141 continue 

142 import json 

143 

144 data = json.loads(line) 

145 yield GenerateResponse( 

146 model=data["model"], 

147 prompt=request.prompt, 

148 response=data["response"], 

149 done=data.get("done", False), 

150 context=data.get("context", []), 

151 total_duration=data.get("total_duration", 0), 

152 load_duration=data.get("load_duration", 0), 

153 prompt_eval_count=data.get("prompt_eval_count", 0), 

154 prompt_eval_duration=data.get("prompt_eval_duration", 0), 

155 eval_count=data.get("eval_count", 0), 

156 eval_duration=data.get("eval_duration", 0), 

157 )

203 async def chat(self, request: ChatRequest) -> ChatMessage: 

204 """Generate chat response. 

205 

206 Args: 

207 request: Chat request with messages and parameters. 

208 

209 Returns: 

210 Chat response message from assistant. 

211 

212 Raises: 

213 httpx.HTTPError: If API request fails. 

214 """ 

215 log.info("ollama_chat", model=request.model, messages=len(request.messages)) 

216 

217 messages = [{"role": m.role, "content": m.content} for m in request.messages] 

218 

219 payload: dict[str, Any] = { 

220 "model": request.model, 

221 "messages": messages, 

222 "temperature": request.temperature, 

223 "top_p": request.top_p, 

224 "top_k": request.top_k, 

225 "num_predict": request.num_predict, 

226 "stream": False, 

227 } 

228 

229 response = await self.client.post("/api/chat", json=payload) 

230 response.raise_for_status() 

231 

232 data = response.json() 

233 message = data["message"] 

234 

235 return ChatMessage(role=message["role"], content=message["content"])
