10class RateLimiter:

13 Implements: 

14 - Token bucket algorithm for rate limiting 

15 - Per-user and per-IP rate limits 

16 - Configurable limits and time windows

18 

19 def __init__(self, requests_per_minute: int = 60, burst_size: int | None = None): 

20 """Initialize rate limiter. 

21 

22 Args: 

23 requests_per_minute: Maximum requests per minute 

24 burst_size: Maximum burst size (defaults to requests_per_minute) 

25 """ 

26 self.requests_per_minute = requests_per_minute 

27 self.burst_size = burst_size or requests_per_minute 

28 self.tokens_per_second = requests_per_minute / 60.0 

29 

30 # In-memory storage - sufficient for single-instance deployments 

31 # For distributed systems, use RedisRateLimiter below 

32 # See: docs/DEPLOYMENT.md for production rate limiting setup 

33 self._buckets: dict[str, dict[str, float]] = defaultdict( 

34 lambda: {"tokens": self.burst_size, "last_update": time.time()} 

35 ) 

36 

37 def _refill_bucket(self, bucket_key: str) -> None: 

38 """Refill tokens in bucket based on elapsed time.""" 

39 bucket = self._buckets[bucket_key] 

40 now = time.time() 

41 elapsed = now - bucket["last_update"] 

42 

43 # Add tokens based on time elapsed 

44 tokens_to_add = elapsed * self.tokens_per_second 

45 bucket["tokens"] = min(self.burst_size, bucket["tokens"] + tokens_to_add) 

46 bucket["last_update"] = now 

47 

48 def check_rate_limit(self, key: str) -> tuple[bool, dict[str, int]]: 

49 """Check if request should be allowed. 

50 

51 Args: 

52 key: Unique identifier (user_id, IP, etc.) 

53 

54 Returns: 

55 Tuple of (allowed, limit_info) 

56 """ 

57 self._refill_bucket(key) 

58 bucket = self._buckets[key] 

59 

60 limit_info = { 

61 "limit": self.requests_per_minute, 

62 "remaining": int(bucket["tokens"]), 

63 "reset": int(bucket["last_update"] + 60), 

64 } 

65 

66 if bucket["tokens"] >= 1.0: 

67 bucket["tokens"] -= 1.0 

68 return True, limit_info 

69 else: 

70 return False, limit_info 

71 

72 def reset(self, key: str) -> None: 

73 """Reset rate limit for a key.""" 

74 if key in self._buckets: 

75 del self._buckets[key]
