75def _collect_counter_value(counter: Counter) -> float: 

76 """Return the sum of all samples for the provided counter.""" 

77 

78 total = 0.0 

79 for metric in counter.collect(): 

80 for sample in metric.samples: 

81 if sample.name == counter._name: 

82 total += float(sample.value) 

83 return total
