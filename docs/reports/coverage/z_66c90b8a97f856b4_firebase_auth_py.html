34def init_firebase(credentials_path: Optional[str] = None) -> None: 

35 """Initialize Firebase Admin SDK. 

36 

37 Args: 

38 credentials_path: Path to Firebase service account JSON. 

39 If None, uses GOOGLE_APPLICATION_CREDENTIALS env var. 

40 

41 Raises: 

42 RuntimeError: If Firebase initialization fails. 

43 """ 

44 global _firebase_initialized, _firebase_app 

45 

46 if _firebase_initialized: 

47 return 

48 

49 try: 

50 if credentials_path: 

51 creds = credentials.Certificate(credentials_path) 

52 _firebase_app = firebase_admin.initialize_app(creds) 

53 else: 

54 # Uses GOOGLE_APPLICATION_CREDENTIALS environment variable 

55 _firebase_app = firebase_admin.initialize_app() 

56 

57 logger.info("✅ Firebase initialized successfully") 

58 _firebase_initialized = True 

59 

60 except Exception as e: 

61 logger.error(f"❌ Failed to initialize Firebase: {e}") 

62 raise RuntimeError(f"Firebase initialization failed: {e}") from e

113def _handle_token_verification_error(error: Exception, require_auth: bool) -> dict[str, Any]: 

114 """Handle token verification errors. 

115 

116 Args: 

117 error: The verification error 

118 require_auth: If True, raise HTTPException; if False, return empty dict 

119 

120 Returns: 

121 Empty dict if not require_auth 

122 

123 Raises: 

124 HTTPException: If require_auth=True 

125 """ 

126 if isinstance(error, firebase_auth.ExpiredSignInError): 

127 logger.warning("⚠️ Expired authentication token") 

128 if require_auth: 

129 raise HTTPException( 

130 status_code=status.HTTP_401_UNAUTHORIZED, detail="Token has expired" 

131 ) from None 

132 return {} 

133 

134 if isinstance(error, firebase_auth.RevokedSignInError): 

135 logger.warning("⚠️ Revoked authentication token - user must re-authenticate") 

136 if require_auth: 

137 raise HTTPException( 

138 status_code=status.HTTP_401_UNAUTHORIZED, 

139 detail="Authentication revoked. Please sign in again.", 

140 ) from None 

141 return {} 

142 

143 if isinstance(error, firebase_auth.InvalidIdTokenError): 

144 logger.warning("⚠️ Invalid authentication token") 

145 if require_auth: 

146 raise HTTPException( 

147 status_code=status.HTTP_401_UNAUTHORIZED, detail="Invalid token" 

148 ) from None 

149 return {} 

150 

151 logger.error(f"❌ Token verification failed: {error}") 

152 if require_auth: 

153 raise HTTPException( 

154 status_code=status.HTTP_401_UNAUTHORIZED, 

155 detail="Authentication failed", 

156 ) from error 

157 return {}

160async def get_current_user(request: Request, require_auth: bool = True) -> dict[str, Any]: 

161 """Extract and verify Firebase JWT token from request. 

162 

163 Args: 

164 request: FastAPI request object 

165 require_auth: If True, raise error if auth fails. If False, return empty dict. 

166 

167 Returns: 

168 User claims from verified JWT token 

169 

170 Raises: 

171 HTTPException: 401 if token missing, invalid, or revoked 

172 """ 

173 # Extract token from Authorization header 

174 auth_header = request.headers.get("Authorization", "") 

175 is_valid, token = _extract_token_from_header(auth_header) 

176 

177 if not is_valid: 

178 if require_auth: 

179 raise HTTPException( 

180 status_code=status.HTTP_401_UNAUTHORIZED, 

181 detail="Missing or invalid Authorization header", 

182 ) 

183 return {} 

184 

185 # Check Firebase is initialized 

186 _check_firebase_available(require_auth) 

187 

188 # Verify token 

189 try: 

190 return _verify_firebase_token(token) 

191 

192 except Exception as e: 

193 return _handle_token_verification_error(e, require_auth)

278def get_or_create_user(email: str, display_name: Optional[str] = None) -> dict[str, Any]: 

279 """Get or create Firebase user. 

280 

281 Args: 

282 email: User email address 

283 display_name: Optional display name 

284 

285 Returns: 

286 User data dict 

287 

288 Raises: 

289 RuntimeError: If Firebase not initialized or operation fails 

290 """ 

291 if not _firebase_initialized: 

292 raise RuntimeError("Firebase not initialized") 

293 

294 try: 

295 # Try to get existing user 

296 user = firebase_auth.get_user_by_email(email) 

297 logger.debug(f"✅ Found existing user: {email}") 

298 return {"uid": user.uid, "email": user.email, "display_name": user.display_name}

300 except firebase_auth.UserNotFoundError:

307 except Exception as e:

312 logger.error(f"❌ Failed to get user {email}: {e}")
