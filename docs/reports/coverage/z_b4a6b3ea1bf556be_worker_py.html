41 self.db_manager = db_manager

48 async def start(self) -> None: 

49 """Start the background worker.""" 

50 if self._running: 

51 return 

52 

53 self._running = True 

54 self._task = asyncio.create_task(self._run_loop()) 

55 log.info("Training worker started.") 

56 

57 async def stop(self) -> None: 

58 """Stop the background worker.""" 

59 self._running = False 

60 if self._task: 

61 self._task.cancel() 

62 try: 

63 await self._task 

64 except asyncio.CancelledError: 

65 pass 

66 log.info("Training worker stopped.") 

67 

68 async def _run_loop(self) -> None: 

69 """Main worker loop.""" 

70 while self._running: 

71 try: 

72 await self._process_next_job() 

73 except Exception as e: 

74 log.exception("Error in training worker loop: %s", e) 

75 

76 await asyncio.sleep(self.poll_interval) 

77 

78 async def _process_next_job(self) -> None: 

79 """Pick up and process the next pending job.""" 

80 async with self.db_manager.SessionLocal() as session: 

81 repository = TrainingJobRepository(session) 

82 

83 # 1. Look for pending jobs 

84 pending_jobs = await repository.list_jobs(status=TrainingStatus.PENDING) 

85 if not pending_jobs: 

86 return 

87 

88 job = pending_jobs[0] 

89 job_id = cast(uuid.UUID, job.id) 

90 log.info("Starting processing for job %s: %s", job_id, job.base_model) 

91 

92 # 2. Acquire GPU Training Lock 

93 success = await self.resources.acquire(WorkloadType.TRAINING, timeout=60.0) 

94 if not success: 

95 log.warning( 

96 "Could not acquire GPU lock for training job %s. Retrying later.", job_id 

97 ) 

98 return 

99 

100 try: 

101 # 3. Mark as running 

102 await repository.update_status(job_id, TrainingStatus.TRAINING) 

103 await session.commit() 

104 

105 # 4. Execute (this will block until done) 

106 try: 

107 # Parse config back to pydantic model for type safety in engine 

108 config = TrainingConfig(**cast(dict[str, Any], job.config)) 

109 dataset_path = Path(cast(str, job.dataset_id)) 

110 

111 await self.engine.train( 

112 job_id=str(job_id), 

113 config=config, 

114 dataset_path=dataset_path, 

115 ) 

116 await repository.update_status(job_id, TrainingStatus.COMPLETED) 

117 log.info("Job %s completed successfully.", job_id) 

118 except Exception as e: 

119 log.error("Job %s failed with error: %s", job_id, e) 

120 await repository.update_status(job_id, TrainingStatus.FAILED) 

121 

122 await session.commit() 

123 

124 finally: 

125 # 5. Always release the lock 

126 self.resources.release()
