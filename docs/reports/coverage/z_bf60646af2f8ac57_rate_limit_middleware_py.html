17class RateLimitMiddleware(BaseHTTPMiddleware): 

18 """FastAPI middleware for rate limiting. 

19 

20 Applies rate limits based on: 

21 - User ID (for authenticated requests) 

22 - IP address (for unauthenticated requests) 

23 - API endpoint 

24 """ 

25 

26 def __init__( 

27 self, 

28 app: Any, 

29 requests_per_minute: int = 60, 

30 burst_size: int | None = None, 

31 exclude_paths: list[str] | None = None, 

32 ) -> None: 

33 """Initialize rate limit middleware. 

34 

35 Args: 

36 app: FastAPI application 

37 requests_per_minute: Global rate limit 

38 burst_size: Maximum burst size 

39 exclude_paths: Paths to exclude from rate limiting 

40 """ 

41 super().__init__(app) 

42 self.limiter = RateLimiter(requests_per_minute, burst_size) 

43 self.exclude_paths = exclude_paths or ["/health", "/docs", "/openapi.json"] 

44 

45 async def dispatch( 

46 self, request: Request, call_next: Callable[[Request], Awaitable[Response]] 

47 ) -> Response: 

48 """Process request with rate limiting. 

49 

50 Args: 

51 request: Incoming request 

52 call_next: Next middleware/route handler 

53 

54 Returns: 

55 Response 

56 """ 

57 # Skip rate limiting for excluded paths 

58 if request.url.path in self.exclude_paths: 

59 return await call_next(request) 

60 

61 # Determine rate limit key 

62 # Priority: user_id > api_key > ip_address 

63 rate_limit_key = None 

64 

65 # Check for authenticated user (set by auth middleware) 

66 if hasattr(request.state, "user"): 

67 rate_limit_key = f"user:{request.state.user.id}" 

68 else: 

69 # Fall back to IP address 

70 client_ip = request.client.host if request.client else "unknown" 

71 rate_limit_key = f"ip:{client_ip}" 

72 

73 # Check rate limit 

74 allowed, limit_info = self.limiter.check_rate_limit(rate_limit_key) 

75 

76 if not allowed: 

77 logger.warning(f"Rate limit exceeded for {rate_limit_key} on {request.url.path}") 

78 return JSONResponse( 

79 status_code=status.HTTP_429_TOO_MANY_REQUESTS, 

80 content={ 

81 "error": { 

82 "message": "Rate limit exceeded", 

83 "type": "rate_limit_error", 

84 "status_code": status.HTTP_429_TOO_MANY_REQUESTS, 

85 } 

86 }, 

87 headers={ 

88 "X-RateLimit-Limit": str(limit_info["limit"]), 

89 "X-RateLimit-Remaining": "0", 

90 "X-RateLimit-Reset": str(limit_info["reset"]), 

91 "Retry-After": "60", 

92 }, 

93 ) 

94 

95 # Add rate limit headers to response 

96 response = await call_next(request) 

97 response.headers["X-RateLimit-Limit"] = str(limit_info["limit"]) 

98 response.headers["X-RateLimit-Remaining"] = str(limit_info["remaining"]) 

99 response.headers["X-RateLimit-Reset"] = str(limit_info["reset"]) 

100 

101 return response
