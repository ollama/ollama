41 def __init__( 

42 self, 

43 service_name: str = "ollama-api", 

44 jaeger_host: str = "jaeger", 

45 jaeger_port: int = 6831, 

46 jaeger_udp_port: int = 6831, 

47 trace_sample_rate: float = 0.1, 

48 ): 

49 """ 

50 Initialize Jaeger configuration 

51 

52 Args: 

53 service_name: Service name for traces 

54 jaeger_host: Jaeger agent host 

55 jaeger_port: Jaeger agent port 

56 jaeger_udp_port: Jaeger UDP port (for Thrift protocol) 

57 trace_sample_rate: Fraction of traces to sample (0.0-1.0) 

58 """ 

59 self.service_name = service_name 

60 self.jaeger_host = jaeger_host 

61 self.jaeger_port = jaeger_port 

62 self.jaeger_udp_port = jaeger_udp_port 

63 self.trace_sample_rate = trace_sample_rate 

64 self._tracer_provider: TracerProvider | None = None 

65 

66 def initialize_tracer(self) -> TracerProvider: 

67 """ 

68 Initialize Jaeger tracer provider 

69 

70 Returns: 

71 Configured TracerProvider 

72 """ 

73 try: 

74 # Create Jaeger exporter 

75 jaeger_exporter = JaegerExporter( 

76 agent_host_name=self.jaeger_host, 

77 agent_port=self.jaeger_udp_port, 

78 ) 

79 

80 # Create tracer provider with sampler 

81 sampler = ParentBased(root=TraceIdRatioBased(self.trace_sample_rate)) 

82 trace_provider = TracerProvider(sampler=sampler) 

83 

84 # Add Jaeger exporter 

85 trace_provider.add_span_processor(BatchSpanProcessor(jaeger_exporter)) 

86 

87 # Set global tracer provider 

88 trace.set_tracer_provider(trace_provider) 

89 

90 self._tracer_provider = trace_provider 

91 logger.info(f"✅ Jaeger tracer initialized: {self.jaeger_host}:{self.jaeger_port}") 

92 

93 return trace_provider 

94 

95 except Exception as e: 

96 logger.error(f"❌ Failed to initialize Jaeger tracer: {e}") 

97 raise
