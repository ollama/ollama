89@router.post("/generate") 

90async def generate( 

91 request: GenerateRequest, 

92 manager: OllamaModelManager = Depends(get_model_manager), 

93 resources: ResourceDependency = None, 

94) -> StreamingResponse | GenerateResponse: 

95 """Generate text using specified model. 

96 

97 Supports both streaming and non-streaming responses based on request. 

98 

99 Args: 

100 request: Generation request with model and parameters 

101 manager: Model manager instance 

102 resources: Resource manager for GPU arbitration 

103 

104 Returns: 

105 Text generation response (streaming or complete) 

106 """ 

107 from fastapi import status 

108 

109 from ollama.services.resources.types import WorkloadType 

110 

111 # FAANG-Grade: GPU Resource Arbitration 

112 # Training jobs hold a long-term lock. Inference must wait or fail fast. 

113 if resources and not await resources.acquire(WorkloadType.INFERENCE, timeout=5.0): 

114 log.warning("inference_resource_conflict for model %s", request.model) 

115 raise HTTPException( 

116 status_code=status.HTTP_503_SERVICE_UNAVAILABLE, 

117 detail="GPU resources currently occupied by training job. Try again later.", 

118 ) 

119 

120 try: 

121 ollama_request = OllamaGenerateRequest( 

122 model=request.model, 

123 prompt=request.prompt, 

124 system=request.system, 

125 temperature=request.temperature, 

126 top_p=request.top_p, 

127 top_k=request.top_k, 

128 repeat_penalty=request.repeat_penalty, 

129 num_predict=request.num_predict, 

130 stream=request.stream, 

131 ) 

132 

133 if request.stream: 

134 

135 async def generate_stream() -> AsyncGenerator[str, None]: 

136 """Stream generation responses as Server-Sent Events.""" 

137 async for response in manager.generate(ollama_request): 

138 yield f"data: {json.dumps(dataclasses.asdict(response))}\n\n" 

139 

140 return StreamingResponse( 

141 generate_stream(), 

142 media_type="text/event-stream", 

143 headers={ 

144 "Cache-Control": "no-cache", 

145 "Connection": "keep-alive", 

146 }, 

147 ) 

148 else: 

149 # Non-streaming: collect all and return 

150 full_response = "" 

151 final_data = None 

152 async for response in manager.generate(ollama_request): 

153 full_response += response.response 

154 if response.done: 

155 final_data = response 

156 

157 return GenerateResponse( 

158 model=final_data.model if final_data else request.model, 

159 prompt=request.prompt, 

160 response=full_response, 

161 done=True, 

162 total_duration=final_data.total_duration if final_data else 0, 

163 prompt_eval_count=final_data.prompt_eval_count if final_data else 0, 

164 eval_count=final_data.eval_count if final_data else 0, 

165 ) 

166 

167 except ValueError as e: 

168 log.warning(f"Invalid request: {e!s}") 

169 raise HTTPException(status_code=400, detail=str(e)) from e 

170 except Exception as e: 

171 log.error(f"Generation failed: {e!s}") 

172 raise HTTPException(status_code=500, detail="Generation failed") from e

251@router.post("/chat") 

252async def chat_completion( 

253 request: ConversationRequest, 

254 manager: OllamaModelManager = Depends(get_model_manager), 

255) -> StreamingResponse: 

256 """Chat completion with conversation history. 

257 

258 Supports multi-turn conversations with model context. 

259 

260 Args: 

261 request: Chat request with message history 

262 manager: Model manager instance 

263 

264 Returns: 

265 Chat response (streaming) 

266 """ 

267 try: 

268 # Build prompt from conversation history 

269 system_prompt = "You are a helpful assistant." 

270 conversation_text = "" 

271 

272 for msg in request.messages: 

273 if msg.role == "system": 

274 system_prompt = msg.content 

275 else: 

276 conversation_text += f"{msg.role}: {msg.content}\n" 

277 

278 conversation_text += "assistant: " 

279 

280 ollama_request = OllamaGenerateRequest( 

281 model=request.model, 

282 prompt=conversation_text, 

283 system=system_prompt, 

284 temperature=request.temperature, 

285 stream=request.stream, 

286 ) 

287 

288 async def chat_stream() -> AsyncGenerator[str, None]: 

289 """Stream chat responses as Server-Sent Events.""" 

290 async for response in manager.generate(ollama_request): 

291 yield f"data: {json.dumps(dataclasses.asdict(response))}\n\n" 

292 

293 return StreamingResponse( 

294 chat_stream(), 

295 media_type="text/event-stream", 

296 headers={ 

297 "Cache-Control": "no-cache", 

298 "Connection": "keep-alive", 

299 }, 

300 ) 

301 

302 except Exception as e: 

303 log.error(f"Chat completion failed: {e!s}") 

304 raise HTTPException(status_code=500, detail="Chat completion failed") from e
