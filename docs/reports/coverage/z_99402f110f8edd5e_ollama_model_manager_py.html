36 async def list_available_models(self) -> list[Model]: 

37 """List all available models from Ollama. 

38 

39 Returns: 

40 List of available models.

50 async def get_model(self, name: str) -> Model: 

51 """Get specific model by name. 

52 

53 Args: 

54 name: Model name (e.g., 'llama2', 'mistral'). 

55 

56 Returns: 

57 Model information. 

58 

59 Raises: 

60 ModelNotFoundError: If model not found. 

61 """ 

62 if name in self._model_cache: 

63 return self._model_cache[name] 

64 

65 if name not in self._models: 

66 await self.list_available_models() 

67 

68 if name not in self._models: 

69 log.error("model_not_found", model=name) 

70 raise ModelNotFoundError(f"Model {name} not found") 

71 

72 model = self._models[name] 

73 self._model_cache[name] = model 

74 return model
